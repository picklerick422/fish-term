```javascript
function makeFramer() {

  return { push: (u8) => dec.decode(u8, { stream: true }) };
}

test('decodes ASCII in one frame', () => {

  assert.equal(f.push(new Uint8Array([0x68, 0x69])), 'hi');
```

```typescript
import { util } from '@kit.ArkTS';

// Decodes a stream of UTF-8 byte frames into strings, buffering any trailing
// partial multibyte sequence until the next frame completes it (spec §4.2).
export class Utf8Framer {
  private decoder: util.TextDecoder = util.TextDecoder.create(
    'utf-8', { fatal: false, ignoreBOM: true });

  push(bytes: Uint8Array): string {
    if (bytes.length === 0) {
      return '';
    }
    return this.decoder.decodeWithStream(bytes, { stream: true });
  }

  reset(): void {
    this.decoder = util.TextDecoder.create('utf-8', { fatal: false, ignoreBOM: true });
  }
}
```

```javascript
export function buildWandUrl(config, cols, rows) {
  const scheme = config.tls ? 'wss' : 'ws';
```

```javascript
export class Backoff {
  constructor(base = 1000, max = 30000) { this.base = base; this.max = max; this.attempt = 0; }
  next() {
    const delay = Math.min(this.max, this.base * Math.pow(2, this.attempt));
    this.attempt++;
    return delay;
  }
  reset() { this.attempt = 0; }
}
```

```typescript
// Exponential backoff for reconnects: base, 2x each attempt, capped at max (spec §4.3).
export class Backoff {
  private readonly base: number;
  private readonly max: number;
  private attempt: number = 0;

  constructor(base: number = 1000, max: number = 30000) {
    this.base = base;
    this.max = max;
  }

  next(): number {
    const delay: number = Math.min(this.max, this.base * Math.pow(2, this.attempt));
    this.attempt++;
    return delay;
  }

  reset(): void {
    this.attempt = 0;
  }
}
```

### Task 1.5: `WandWebSocketDriver`

**Files:** Create `entry/src/main/ets/transport/WandWebSocketDriver.ets`

Protocol (verified against `/tmp/wand-agent-src`):
- Connect `ws[s]://host:port/ws?token&cols&rows[&cwd&shell]`. Bad token → HTTP 401 (surfaces as connect error).
- Server → client: **binary** frame = PTY output; **text** JSON = `{"type":"cwd","dir"}`, `{"type":"error","error"}`, `{"type":"shells","list"}`, `{"type":"forked","id"}`.
- Client → server: **binary** = stdin; **text** JSON = `{"type":"resize","cols","rows"}`, `{"type":"ping","ts"}`.
- **No pong is ever sent** — heartbeat is fire-and-forget; detect disconnect via `close`/`error` only.

- [ ] **Step 1: Write the driver**

```typescript
import { webSocket } from '@kit.NetworkKit';
import { BusinessError } from '@kit.BasicServicesKit';
import { util } from '@kit.ArkTS';
import { DriverState, DriverStatus, TerminalDriver, WandConnectConfig } from './TerminalDriver';
import { Utf8Framer } from './Utf8Framer';
import { buildWandUrl } from './WandUrl';
import { Backoff } from './Backoff';

const HEARTBEAT_MS: number = 15000;

interface ServerCtrl {
  type: string;
  dir?: string;
  error?: string;
}

export class WandWebSocketDriver implements TerminalDriver {
  private readonly config: WandConnectConfig;
  private ws: webSocket.WebSocket | null = null;
  private readonly framer: Utf8Framer = new Utf8Framer();
  private readonly encoder: util.TextEncoder = new util.TextEncoder();
  private readonly backoff: Backoff = new Backoff(1000, 30000);

  private outputCb: ((data: string) => void) | null = null;
  private statusCb: ((s: DriverStatus) => void) | null = null;
  private cwdCb: ((dir: string) => void) | null = null;

  private cols: number = 80;
  private rows: number = 24;
  private connected: boolean = false;
  private manualStop: boolean = false;
  private reconnectTimer: number = -1;
  private heartbeatTimer: number = -1;

  constructor(config: WandConnectConfig) {
    this.config = config;
  }

  onOutput(cb: (data: string) => void): void {
    this.outputCb = cb;
  }

  onStatus(cb: (s: DriverStatus) => void): void {
    this.statusCb = cb;
  }

  // Optional: server pushes cwd updates; used for tab titles in later phases.
  onCwd(cb: (dir: string) => void): void {
    this.cwdCb = cb;
  }

  start(cols: number, rows: number): void {
    this.cols = Math.max(1, cols);
    this.rows = Math.max(1, rows);
    this.manualStop = false;
    this.backoff.reset();
    this.connect();
  }

  write(data: string): void {
    if (!this.connected || !this.ws || data.length === 0) {
      return;
    }
    const bytes: Uint8Array = this.encoder.encodeInto(data);
    const buffer: ArrayBuffer = bytes.buffer.slice(
      bytes.byteOffset, bytes.byteOffset + bytes.byteLength);
    this.ws.send(buffer).catch((err: BusinessError) => {
      console.error('wand send failed: ' + JSON.stringify(err));
    });
  }

  resize(cols: number, rows: number): void {
    this.cols = Math.max(1, cols);
    this.rows = Math.max(1, rows);
    if (this.connected && this.ws) {
      this.sendText(`{"type":"resize","cols":${this.cols},"rows":${this.rows}}`);
    }
  }

  stop(): void {
    this.manualStop = true;
    this.clearReconnect();
    this.stopHeartbeat();
    this.connected = false;
    if (this.ws) {
      const closing: webSocket.WebSocket = this.ws;
      this.ws = null;
      closing.close().catch((err: BusinessError) => {
        console.error('wand close failed: ' + JSON.stringify(err));
      });
    }
    this.framer.reset();
    this.emit(DriverState.Closed);
  }

  private connect(): void {
    this.clearReconnect();
    this.framer.reset();
    const url: string = buildWandUrl(this.config, this.cols, this.rows);
    const ws: webSocket.WebSocket = webSocket.createWebSocket();
    this.ws = ws;
    this.emit(DriverState.Connecting);

    ws.on('open', () => {
      this.connected = true;
      this.backoff.reset();
      this.emit(DriverState.Connected);
      this.startHeartbeat();
    });

    ws.on('message', (_err: BusinessError, value: string | ArrayBuffer) => {
      this.onMessage(value);
    });

    ws.on('close', (_err: BusinessError, _info: webSocket.CloseResult) => {
      this.connected = false;
      this.stopHeartbeat();
      if (this.manualStop) {
        this.emit(DriverState.Closed);
        return;
      }
      this.emit(DriverState.Reconnecting);
      this.scheduleReconnect();
    });

    ws.on('error', (err: BusinessError) => {
      this.emit(DriverState.Error, 'connection error (' + err.code + ')');
    });

    ws.connect(url, (err: BusinessError, _ok: boolean) => {
      if (err) {
        this.emit(DriverState.Error, 'connect failed (' + err.code + ')');
        if (!this.manualStop) {
          this.scheduleReconnect();
        }
      }
    });
  }

  private onMessage(value: string | ArrayBuffer): void {
    if (typeof value === 'string') {
      this.handleCtrl(value);
      return;
    }
    const text: string = this.framer.push(new Uint8Array(value));
    if (text.length > 0 && this.outputCb) {
      this.outputCb(text);
    }
  }

  private handleCtrl(text: string): void {
    let ctrl: ServerCtrl;
    try {
      ctrl = JSON.parse(text) as ServerCtrl;
    } catch (err) {
      console.error('wand bad ctrl frame: ' + text);
      return;
    }
    if (ctrl.type === 'error') {
      this.emit(DriverState.Error, ctrl.error ?? 'server error');
    } else if (ctrl.type === 'cwd') {
      if (this.cwdCb && ctrl.dir) {
        this.cwdCb(ctrl.dir);
      }
    }
    // 'shells' / 'forked' ignored in v1.
  }

  private sendText(json: string): void {
    if (!this.ws) {
      return;
    }
    this.ws.send(json).catch((err: BusinessError) => {
      console.error('wand sendText failed: ' + JSON.stringify(err));
    });
  }

  private startHeartbeat(): void {
    this.stopHeartbeat();
    this.heartbeatTimer = setInterval(() => {
      if (this.connected) {
        this.sendText(`{"type":"ping","ts":${Date.now()}}`);
      }
    }, HEARTBEAT_MS) as number;
  }

  private stopHeartbeat(): void {
    if (this.heartbeatTimer >= 0) {
      clearInterval(this.heartbeatTimer);
      this.heartbeatTimer = -1;
    }
  }

  private scheduleReconnect(): void {
    if (this.manualStop || this.reconnectTimer >= 0) {
      return;
    }
    const delay: number = this.backoff.next();
    this.reconnectTimer = setTimeout(() => {
      this.reconnectTimer = -1;
      if (!this.manualStop) {
        this.connect();
      }
    }, delay) as number;
  }

  private clearReconnect(): void {
    if (this.reconnectTimer >= 0) {
      clearTimeout(this.reconnectTimer);
      this.reconnectTimer = -1;
    }
  }

  private emit(state: DriverState, message?: string): void {
    if (this.statusCb) {
      this.statusCb({ state, message });
    }
  }
}
```

```typescript
import { common } from '@kit.AbilityKit';
import { TerminalController, TerminalSize } from 'libghostty_ohos';
import { DriverStatus, TerminalDriver } from '../transport/TerminalDriver';

// Owns one controller + one driver and wires the spec §3.3 dataflow.
export class TerminalSession {
  readonly controller: TerminalController = new TerminalController();
  private readonly driver: TerminalDriver;
  private context: common.UIAbilityContext | null = null;
  private statusCb: ((s: DriverStatus) => void) | null = null;

  private started: boolean = false;
  private startRetryTimer: number = -1;
  private lastCols: number = 0;
  private lastRows: number = 0;

  constructor(driver: TerminalDriver) {
    this.driver = driver;
    this.driver.onOutput((data: string): void => {
      this.controller.feed(data);
    });
    this.driver.onStatus((s: DriverStatus): void => {
      if (this.statusCb) {
        this.statusCb(s);
      }
    });
  }

  onStatus(cb: (s: DriverStatus) => void): void {
    this.statusCb = cb;
  }

  attach(context: common.UIAbilityContext): void {
    this.context = context;
    this.controller.setInputListener((data: string): void => {
      this.driver.write(data);
    });
    this.controller.setAttachmentListener((attached: boolean): void => {
      if (attached) {
        this.startWhenSized();
      } else {
        this.stop();
      }
    });
    this.controller.setSizeListener((size: TerminalSize): void => {
      if (!this.started) {
        return;
      }
      const cols: number = Math.max(1, size.cols || 0);
      const rows: number = Math.max(1, size.rows || 0);
      if (cols === this.lastCols && rows === this.lastRows) {
        return;
      }
      this.lastCols = cols;
      this.lastRows = rows;
      this.driver.resize(cols, rows);
    });
  }

  detach(): void {
    this.controller.setInputListener(null);
    this.controller.setAttachmentListener(null);
    this.controller.setSizeListener(null);
    this.stop();
    this.context = null;
  }

  private startWhenSized(): void {
    if (this.started) {
      return;
    }
    const size: TerminalSize = this.controller.getTerminalSize();
    const cols: number = Math.max(0, size.cols || 0);
    const rows: number = Math.max(0, size.rows || 0);
    if (cols <= 0 || rows <= 0) {
      if (this.startRetryTimer < 0) {
        this.startRetryTimer = setTimeout((): void => {
          this.startRetryTimer = -1;
          this.startWhenSized();
        }, 50) as number;
      }
      return;
    }
    this.started = true;
    this.lastCols = cols;
    this.lastRows = rows;
    this.driver.start(cols, rows);
  }

  private stop(): void {
    if (this.startRetryTimer >= 0) {
      clearTimeout(this.startRetryTimer);
      this.startRetryTimer = -1;
    }
    if (this.started) {
      this.driver.stop();
      this.started = false;
    }
  }
}
```

```typescript
function stateLabel(s: DriverStatus): string {
  switch (s.state) {
    case DriverState.Connecting: return 'Connecting…';
    case DriverState.Connected: return 'Connected';
    case DriverState.Reconnecting: return 'Reconnecting…';
    case DriverState.Closed: return 'Closed';
    case DriverState.Error: return 'Error: ' + (s.message ?? '');
    default: return 'Idle';
  }
}

  @State private token: string = '';
  @State private tls: boolean = false;
  @State private statusText: string = 'Idle';
  private session: TerminalSession | null = null;
  private context: common.UIAbilityContext | null = null;

    this.context = this.getUIContext().getHostContext() as common.UIAbilityContext;
```
